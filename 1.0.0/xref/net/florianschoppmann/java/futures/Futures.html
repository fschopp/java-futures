24      /**
25       * Returns a new {@link CompletableFuture} that will normally be completed with the list of results of all given
26       * completion stages. The future will fail early if any of the given stages are completed exceptionally.
27       *
28       * <p>If any of the given completion stages completes exceptionally, then the returned future is completed with a
29       * {@link CompletionException}. This is either the first encountered exception (in the order of the given
30       * {@link Iterable}) if that is a {@link CompletionException}, or otherwise a fresh {@link CompletionException}
31       * holding the first encountered exception as cause.
32       *
33       * <p>Unlike {@link #collect(Iterable)}, the returned future may be completed before all completion stages are
34       * completed. Specifically, in case of exceptional completion of the returned future, some of the given completion
35       * stages may not yet be completed.
36       *
37       * <p>The order of the given completion stages is preserved in the result list.
38       *
39       * @param completionStages completion stages supplying the elements of the list that the returned future will be
40       *     completed with
41       * @param <T> the element type of the list that the new future will be completed with
42       * @return the new future
43       */
44      public static <T> CompletableFuture<List<T>> shortCircuitCollect(
45              Iterable<? extends CompletionStage<? extends T>> completionStages) {
46          Objects.requireNonNull(completionStages);
47  
48          CompletableFuture<List<T>> listFuture = CompletableFuture.completedFuture((List<T>) new ArrayList<T>());
49          for (CompletionStage<? extends T> completionStage: completionStages) {
50              CompletableFuture<List<T>> newListFuture = new CompletableFuture<>();
51              listFuture.whenComplete((list, listThrowable) -> {
52                  assert list != null || listThrowable != null;
53                  if (listThrowable != null) {
54                      newListFuture.completeExceptionally(listThrowable);
55                  } else {
56                      completionStage.whenComplete((element, throwable) -> {
57                          if (throwable != null) {
58                              newListFuture.completeExceptionally(encodeException(throwable));
59                          } else {
60                              list.add(element);
61                              newListFuture.complete(list);
62                          }
63                      });
64                  }
65              });
66              listFuture = newListFuture;
67          }
68          return listFuture;
69      }

71      /**
72       * Returns a new {@link CompletableFuture} that will normally be completed with a list of results of all given
73       * completion stages.
74       *
75       * <p>If any of the given completion stages completes exceptionally, then the returned future is completed with a
76       * {@link CompletionException}. This is either the first encountered exception (in the order of the given
77       * {@link Iterable}) if that is a {@link CompletionException}, or otherwise a fresh {@link CompletionException}
78       * holding the first encountered exception as cause.
79       *
80       * <p>Similar to {@link CompletableFuture#allOf(CompletableFuture[])}, this method guarantees that the returned
81       * future will only be completed <em>after</em> all given completion stages have completed.
82       *
83       * <p>The order of the given completion stages is preserved in the result list.
84       *
85       * @param completionStages completion stages supplying the elements of the list that the returned future will be
86       *     completed with
87       * @param <T> the element type of the list that the new future will be completed with
88       * @return future that will be completed with a list of the values that the given futures will be completed with
89       */
90      public static <T> CompletableFuture<List<T>> collect(
91              Iterable<? extends CompletionStage<? extends T>> completionStages) {
92          Objects.requireNonNull(completionStages);
93  
94          CompletableFuture<List<T>> listFuture = CompletableFuture.completedFuture((List<T>) new ArrayList<T>());
95          for (CompletionStage<? extends T> completionStage: completionStages) {
96              CompletableFuture<List<T>> newListFuture = new CompletableFuture<>();
97              listFuture.whenComplete((list, listThrowable) -> {
98                  assert list != null || listThrowable != null;
99                  completionStage.whenComplete((element, throwable) -> {
100                     if (throwable != null) {
101                         if (listThrowable != null) {
102                             newListFuture.completeExceptionally(listThrowable);
103                         } else {
104                             newListFuture.completeExceptionally(encodeException(throwable));
105                         }
106                     } else {
107                         if (listThrowable != null) {
108                             newListFuture.completeExceptionally(listThrowable);
109                         } else {
110                             list.add(element);
111                             newListFuture.complete(list);
112                         }
113                     }
114                 });
115             });
116             listFuture = newListFuture;
117         }
118         return listFuture;
119     }

139     private static CompletionException rewrap(CompletionException exception) {
140         Throwable cause = exception.getCause();
141         return cause instanceof WrappedException
142             ? new CompletionException(cause.getCause())
143             : exception;

146     /**
147      * Returns a new {@link CompletableFuture} that is asynchronously completed by a task running in the given executor
148      * with the value obtained by calling the given {@link CheckedSupplier}.
149      *
150      * <p>Just like {@link CompletableFuture#supplyAsync(Supplier, Executor)}, the returned future will be completed
151      * with a {@link CompletionException} if the given supplier throws.
152      *
153      * @param supplier a function returning the value to be used to complete the returned {@link CompletableFuture}
154      * @param executor the executor to use for asynchronous execution
155      * @param <T> the function's return type
156      * @return the new future
157      *
158      * @see CompletableFuture#supplyAsync(Supplier, Executor)
159      */
160     public static <T> CompletableFuture<T> supplyAsync(CheckedSupplier<T> supplier, Executor executor) {
161         CompletionStage<T> completionStage = CompletableFuture.supplyAsync(() -> {
162             try {
163                 return supplier.checkedGet();
164             } catch (Exception exception) {
165                 throw new WrappedException(exception);
166             }
167         }, executor);
168 
169         CompletableFuture<T> future = new CompletableFuture<>();
170         completionStage.whenComplete((result, failure) -> {
171             if (failure != null) {
172                 assert failure instanceof CompletionException;
173                 future.completeExceptionally(rewrap((CompletionException) failure));
174             } else {
175                 future.complete(result);
176             }
177         });
178         return future;
179     }

212     /**
213      * Returns a new {@link CompletableFuture} that will be completed in the same way as the given stage. However, in
214      * case of exceptional completion with a {@link CompletionException}, the returned future will be exceptionally
215      * completed with the <em>cause</em> held in the {@link CompletionException}.
216      *
217      * @param completionStage the completion stage that may be completed with a {@link CompletionException}
218      * @param <T> the type of result of the completion stage
219      * @return the new future
220      */
221     public static <T> CompletableFuture<T> unwrapCompletionException(CompletionStage<T> completionStage) {
222         CompletableFuture<T> future = new CompletableFuture<>();
223         completionStage.whenComplete((value, failure) -> {
224             if (failure instanceof CompletionException) {
225                 @Nullable Throwable cause = failure.getCause();
226                 if (cause != null) {
227                     future.completeExceptionally(cause);
228                 } else {
229                     future.completeExceptionally(failure);
230                 }
231             } else if (failure != null) {
232                 future.completeExceptionally(failure);
233             } else {
234                 future.complete(value);
235             }
236         });
237         return future;
238     }

297     private static <T, U> CompletableFuture<U> thenApplyInternal(
298             Function<BiConsumer<? super T, ? super Throwable>, CompletionStage<T>> whenCompleteFunction,
299             CheckedFunction<? super T, ? extends U> checkedFunction) {
300         return transformCompletionStage(
301             whenCompleteFunction,
302             (result, failure, future) -> {
303                 if (failure == null) {
304                     try {
305                         future.complete(checkedFunction.checkedApply(result));
306                     } catch (Exception exception) {
307                         throw new WrappedException(exception);
308                     }
309                 }
310             }
311         );

392     private static <T, U> CompletableFuture<U> thenComposeInternal(
393             Function<BiConsumer<? super T, ? super Throwable>, CompletionStage<T>> whenCompleteStage,
394             CheckedFunction<? super T, ? extends CompletionStage<U>> checkedFunction) {
395         return transformCompletionStage(
396             whenCompleteStage,
397             (result, failure, future) -> {
398                 if (failure == null) {
399                     try {
400                         CompletionStage<U> nextStage = Objects.requireNonNull(checkedFunction.checkedApply(result));
401                         nextStage.whenComplete((nextResult, nextFailure) -> {
402                             if (nextFailure != null) {
403                                 future.completeExceptionally(encodeException(nextFailure));
404                             } else {
405                                 future.complete(nextResult);
406                             }
407                         });
408                     } catch (Exception exception) {
409                         throw new WrappedException(exception);
410                     }
411                 }
412             }
413         );
414     }

459     private static <T> CompletableFuture<T> whenCompleteInternal(
460             Function<BiConsumer<? super T, ? super Throwable>, CompletionStage<T>> whenCompleteStage,
461             CheckedBiConsumer<? super T, ? super Throwable> action) {
462         return transformCompletionStage(
463             whenCompleteStage,
464             (result, failure, future) -> {
465                 try {
466                     action.checkedAccept(result, failure);
467                     if (failure == null) {
468                         future.complete(result);
469                     }
470                 } catch (Exception exception) {
471                     throw new WrappedException(exception);
472                 }
473             }
474         );
475     }

477     /**
478      * Returns a new {@link CompletableFuture} that is normally completed with the same result as the given stage. If
479      * the given stage completed exceptionally, the given function is called with the exception as argument and the
480      * returned future is completed exceptionally with the result.
481      *
482      * <p>The exception returned by {@code function} is <em>not</em> wrapped in a {@link CompletionException}. However,
483      * if an exception <em>E</em> occurs in the given function, the returned future will be completed exceptionally with
484      * a {@link CompletionException} containing <em>E</em> as cause.
485      *
486      * @param completionStage the completion stage providing the exception for the given function
487      * @param function the exception-translation function
488      * @param <T> the type of the completion stage
489      * @return the new future
490      */
491     public static <T> CompletableFuture<T> translateException(CompletionStage<T> completionStage,
492             Function<Throwable, Throwable> function) {
493         CompletableFuture<T> future = new CompletableFuture<>();
494         completionStage
495             .handle((result, failure) -> {
496                 if (failure != null) {
497                     future.completeExceptionally(Objects.requireNonNull(function.apply(failure)));
498                 } else {
499                     future.complete(result);
500                 }
501                 return null;
502             })
503             .exceptionally(future::completeExceptionally);
504         return future;
505     }

519     /**
520      * Returns a new {@link CompletableFuture} that is normally completed by the given {@code resultConsumer}, and that
521      * is exceptionally completed if a failure occurs.
522      *
523      * <p>This method factors out common code in order to implement <em>thenApply</em>, <em>thenCompose</em>, etc.
524      * methods that that have corresponding methods in {@link CompletionStage}. However, the static methods in this
525      * class allow transformations that throw checked exceptions.
526      */
527     static <T, U> CompletableFuture<U> transformCompletionStage(
528         Function<BiConsumer<? super T, ? super Throwable>, CompletionStage<T>> whenCompleteFunction,
529         TriConsumer<? super T, Throwable, CompletableFuture<U>> resultConsumer
530     ) {
531         CompletableFuture<U> future = new CompletableFuture<>();
532         whenCompleteFunction
533             .apply((result, failure) -> resultConsumer.accept(result, failure, future))
534             .exceptionally(exception -> {
535                 // Assuming that whenCompleteFunction is indeed method whenComplete or whenCompleteAsync, then we expect
536                 // a CompletionException here. However, we are extra cautious here, because a failed assertion or a
537                 // ClassCastException here could potentially create a debugging nightmare (it would be caught in
538                 // CompletableFuture#exceptionally() and then ignored).
539                 future.completeExceptionally(
540                     exception instanceof CompletionException
541                         ? rewrap((CompletionException) exception)
542                         : exception
543                 );
544                 // Result of the new completion stage created by exceptionally() is ignored
545                 return null;
546             });
547         return future;
548     }

550     /**
551      * Returns a new {@link CompletableFuture} that, when the given stage completes normally, is executed using the
552      * supplied Executor, with the given stage's result as the argument to the supplied function. See the
553      * {@link CompletionStage} documentation for rules covering exceptional completion.
554      *
555      * <p>This method behaves like {@link #thenApplyAsync(CompletionStage, CheckedFunction, Executor)}, but it ensures
556      * that the resource passed to {@code function} is properly closed, even if an exception occurs at any stage. This
557      * method may thus be regarded as an asynchronous try-with-resources implementation (with just one resource: The
558      * {@code R} instance that {@code resourceStage} will is completed with).
559      *
560      * <p>Similar to try-with-resources, if an exception occurs after the {@link AutoCloseable} resource has been
561      * opened, then any further exception while automatically closing the resource will be added as suppressed
562      * exception. Suppressed exceptions are added to the original exception, not the {@link CompletionException} that
563      * the returned future would be completed with and that contains the original exception as cause.
564      *
565      * @param resourceStage completion stage that will be completed with an {@link AutoCloseable}
566      * @param function function computing the result for the new future
567      * @param executor the executor to use for asynchronous execution
568      * @param <R> the type of the resource returned by the given completion stage
569      * @param <T> the type of the value returned by the given function
570      * @return the new future
571      */
572     public static <R extends AutoCloseable, T> CompletableFuture<T> thenApplyWithResourceAsync(
573             CompletionStage<R> resourceStage, CheckedFunction<? super R, ? extends T> function,
574             Executor executor) {
575         return thenApplyAsync(resourceStage, resource -> {
576             try (R ignored = resource) {
577                 return function.checkedApply(resource);
578             }
579         }, executor);
580     }

582     /**
583      * Returns a new {@link CompletableFuture} that, when the given stage completes normally, is executed with that
584      * stage as the argument to the supplied function. See the {@link CompletionStage} documentation for rules covering
585      * exceptional completion.
586      *
587      * <p>This method behaves like {@link #thenCompose(CompletionStage, CheckedFunction)}, but it ensures that the
588      * resource passed to {@code function} is properly closed, even if an exception occurs at any stage. This method may
589      * thus be regarded as an asynchronous try-with-resources implementation (with just one resource: The {@code R}
590      * instance that {@code resourceStage} will is completed with).
591      *
592      * <p>Similar to try-with-resources, if an exception occurs after the {@link AutoCloseable} resource has been
593      * opened, then any further exception while automatically closing the resource will be added as suppressed
594      * exception. Suppressed exceptions are added to the original exception, not the {@link CompletionException} that
595      * the returned future would be completed with and that contains the original exception as cause.
596      *
597      * @param resourceStage completion stage that will be completed with an {@link AutoCloseable}
598      * @param function Function returning a completion stage that will compute the result for the new future.
599      * @param <R> the type of the resource returned by the given completion stage
600      * @param <T> the result type of the future returned by the given function
601      * @return the new future
602      */
603     public static <R extends AutoCloseable, T> CompletableFuture<T> thenComposeWithResource(
604             CompletionStage<R> resourceStage,
605             CheckedFunction<? super R, ? extends CompletionStage<T>> function) {
606         return thenCompose(resourceStage, resource -> {
607             try (CloseableWrapper closeableWrapper = new CloseableWrapper(resource)) {
608                 CompletionStage<T> nextStage = Objects.requireNonNull(function.checkedApply(resource));
609                 closeableWrapper.ignoreClose = true;
610                 return whenComplete(nextStage, (result, failure) -> {
611                     // According to JLS8 14.20.3.1, if a primary exception has occurred, then whatever exception occurs
612                     // during close() is added as suppressed exception, even in case of a Throwable that is not an
613                     // Exception.
614                     if (resource != null) {
615                         if (failure != null) {
616                             try {
617                                 resource.close();
618                             } catch (Throwable throwable) {
619                                 decode(failure).addSuppressed(throwable);
620                             }
621                         } else {
622                             resource.close();
623                         }
624                     }
625                 });
626             }
627         });
628     }
629 
630     private static final class CloseableWrapper implements AutoCloseable {
631         @Nullable private final AutoCloseable closeable;
632 
633         /**
634          * No need to synchronize access to this field. The read from {@link #close()} here will always happen either
635          * before or after it is written to, never concurrent to a write in
636          * {@link #thenComposeWithResource(CompletionStage, CheckedFunction)}. However, the field has to be volatile
637          * for cache coherence.
638          */
639         private volatile boolean ignoreClose = false;
640 
641         private CloseableWrapper(@Nullable AutoCloseable closeable) {
642             this.closeable = closeable;
643         }
644 
645         @Override
646         public void close() throws Exception {
647             if (closeable != null && !ignoreClose) {
648                 closeable.close();
649             }
650         }
651     }
